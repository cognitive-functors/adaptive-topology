```python
import openai

def classify_to_c4(text: str) -> tuple[int, int, int]:
 """
 –ö–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ—Ç —Ç–µ–∫—Å—Ç –≤ C4-–ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–æ.
 –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç (T, D, A) ‚àà {0,1,2}¬≥
 """
 prompt = f"""
 Classify this thought into C4 coordinates:

 Text: "{text}"

 T (Time): 0=Past, 1=Present, 2=Future
 D (Scale): 0=Concrete, 1=Abstract, 2=Meta
 A (Agency): 0=Self, 1=Other, 2=System

 Return only: T,D,A
 """

 response = openai.ChatCompletion.create(
 model="gpt-4",
 messages=[{"role": "user", "content": prompt}],
 temperature=0
 )

 t, d, i = map(int, response.choices[0].message.content.split(','))
 return (t, d, i)

# –ü—Ä–∏–º–µ—Ä:
text_A = "–Ø –ø–æ–º–Ω—é, –∫–∞–∫ –æ–Ω –æ–±–∏–¥–µ–ª –º–µ–Ω—è –≤—á–µ—Ä–∞ –Ω–∞ –≤—Å—Ç—Ä–µ—á–µ"

state_B = classify_to_c4(text_B) # ‚Üí (2, 1, 2) = F‚ü®Future, Abstract, System‚ü©

def hamming_distance(s1, s2):
 return sum(a != b for a, b in zip(s1, s2))

gap = hamming_distance(state_A, state_B) # ‚Üí 3
print(f"Cognitive gap: {gap}/3 (maximum)")
```
